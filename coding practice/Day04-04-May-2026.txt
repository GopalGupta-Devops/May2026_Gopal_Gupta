
Day 04- Basic CI/CD Concept with Jenkins % Github

	Jenkins - Basic CI Pipeline
	pipeline {
	  agent any	
	  stages{
	      stages('Build') {
		 steps {
		     echo 'Building project'
		}
	     }
	     stages(s'Test') {
		steps {
		   echo 'Running test'
		}
	     }
	     stages(Deploy) {
	        steps {
	            echo 'Deploying application'
                 }
	     }
          }
       }



       Gitlab CI/CD- Basic Pipeline
      stages:
	- build
	- test
	- deploy

      build:
	stage: build
	script:
	  - echo "Building project"

      test:
     	stage: test
	script:
          - echo "Running test"

      deploy: 
        stage: deploy
	script:
	  - echo "Deploying application"

 
      Ansible- Install and configure Ngnix

      ----
      - name: Install and configure Nginx
       hosts: all
	become: yes	 
	tasks:
          - name: Install nginx
            yum:
              name: nginx
              state: present
 	  - name: Configure nginx
            template:
               src: nginx.conf.j2
               dest: /etc/nginx/nginx.conf
	  - name: Start nginx services
            service:
              name: nginx
	      state: started
              enabled: yes
